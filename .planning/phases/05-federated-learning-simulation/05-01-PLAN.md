Create backend/ml/fl_train.py with the following complete implementation.

CRITICAL constraints:
- NO flwr or ray imports (incompatible with Windows+Python 3.13)
- Use sklearn LogisticRegression ONLY (XGBoost coef_ cannot be FedAvg'd)
- This is synchronous code — called via ThreadPoolExecutor in ws.py
- Pre-fit client models with 2 dummy samples BEFORE assigning coef_/intercept_ (sklearn requirement)
- warm_start=True, max_iter=1, solver="saga" on all LR instances

REGION_NODE_IDS splits 24 nodes by lat: North (>=41.91), Central (41.86-41.91), South (<41.86).
Load ranges make data genuinely non-IID: North 30-80% load (~10% anomaly), Central 40-90% (~15%), South 50-100% (~22%).
Client sizes: North=800, Central=600, South=1000 samples.

```python
"""
AegisFlow Federated Learning simulation — manual FedAvg, no Flower/Ray.
Geographic partitions (Chicago 24-node grid):
  Client 0 North Utility  (lat >= 41.91): generators+substations, ~10% anomaly
  Client 1 Central Utility (41.86-41.91): mixed types, ~15% anomaly
  Client 2 South Utility  (lat < 41.86): high-load nodes, ~22% anomaly
"""
import random
import numpy as np
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score
import joblib
from pathlib import Path
from backend.state import NODE_STATES

FL_MODEL_PATH = Path(__file__).parent / "fl_model.pkl"

FL_PROGRESS: dict = {
    "round": 0, "num_rounds": 10, "running": False,
    "clients": [
        {"id": 0, "name": "North Utility", "history": []},
        {"id": 1, "name": "Central Utility", "history": []},
        {"id": 2, "name": "South Utility", "history": []},
    ],
    "global_history": [],
}

REGION_NODE_IDS: dict[int, list[str]] = {
    0: [nid for nid, ns in NODE_STATES.items() if ns.lat >= 41.91],
    1: [nid for nid, ns in NODE_STATES.items() if 41.86 <= ns.lat < 41.91],
    2: [nid for nid, ns in NODE_STATES.items() if ns.lat < 41.86],
}

N_FEATURES = 9
WINDOW = 5
RANDOM_SEED = 42

_REGION_VOLTAGE_NOMINALS: dict[int, list[float]] = {
    0: [345.0, 138.0],
    1: [138.0, 69.0, 34.0],
    2: [345.0, 69.0, 34.0],
}
_REGION_LOAD_RANGE: dict[int, tuple[float, float]] = {
    0: (30.0, 80.0),
    1: (40.0, 90.0),
    2: (50.0, 100.0),
}


def generate_partition_data(region_id: int, n_samples: int = 600, seed_offset: int = 0) -> tuple[np.ndarray, np.ndarray]:
    rng = random.Random(RANDOM_SEED + seed_offset)
    voltage_nominals = _REGION_VOLTAGE_NOMINALS[region_id]
    load_lo, load_hi = _REGION_LOAD_RANGE[region_id]
    X_rows: list[list[float]] = []
    y_rows: list[int] = []
    for _ in range(n_samples):
        nominal_v = rng.choice(voltage_nominals)
        voltage_base = nominal_v * rng.uniform(0.95, 1.05)
        frequency_base = rng.uniform(59.0, 61.0)
        load_base = rng.uniform(load_lo, load_hi)
        voltages = [voltage_base + rng.uniform(-0.5, 0.5) for _ in range(WINDOW)]
        frequencies = [max(59.0, min(61.0, frequency_base + rng.uniform(-0.05, 0.05))) for _ in range(WINDOW)]
        loads = [max(0.0, min(100.0, load_base + rng.uniform(-2.0, 2.0))) for _ in range(WINDOW)]
        v, f, l = voltages[-1], frequencies[-1], loads[-1]
        row = [v, f, l, float(np.mean(voltages)), float(np.std(voltages)),
               float(np.mean(frequencies)), float(np.std(frequencies)),
               float(np.mean(loads)), float(np.std(loads))]
        label = 1 if l > 94.0 or abs(f - 60.0) > 0.8 else 0
        X_rows.append(row)
        y_rows.append(label)
    return np.array(X_rows, dtype=np.float32), np.array(y_rows, dtype=np.int32)


def _fedavg_params(client_params: list[tuple[np.ndarray, np.ndarray]], client_sizes: list[int]) -> tuple[np.ndarray, np.ndarray]:
    total = sum(client_sizes)
    weights = [s / total for s in client_sizes]
    avg_coef = np.average([p[0] for p in client_params], axis=0, weights=weights)
    avg_intercept = np.average([p[1] for p in client_params], axis=0, weights=weights)
    return avg_coef, avg_intercept


def _make_client_model(global_coef: np.ndarray, global_intercept: np.ndarray) -> LogisticRegression:
    model = LogisticRegression(warm_start=True, max_iter=1, solver="saga", random_state=RANDOM_SEED)
    model.fit(np.zeros((2, N_FEATURES)), [0, 1])  # init coef_ shape
    model.coef_ = global_coef.copy()
    model.intercept_ = global_intercept.copy()
    return model


def run_fl_simulation() -> None:
    """Run 10 FL rounds. Synchronous/CPU-bound. Must be called via ThreadPoolExecutor."""
    NUM_ROUNDS = 10
    PARTITION_SIZES = [800, 600, 1000]
    CLIENT_NAMES = ["North Utility", "Central Utility", "South Utility"]

    FL_PROGRESS["running"] = True
    FL_PROGRESS["round"] = 0
    FL_PROGRESS["global_history"] = []
    for i, c in enumerate(FL_PROGRESS["clients"]):
        c["history"] = []
        c["name"] = CLIENT_NAMES[i]

    partitions = [generate_partition_data(i, PARTITION_SIZES[i], i * 100) for i in range(3)]
    splits = [(X[:int(len(X)*0.8)], X[int(len(X)*0.8):], y[:int(len(y)*0.8)], y[int(len(y)*0.8):]) for X, y in partitions]

    global_model = LogisticRegression(warm_start=True, max_iter=1, solver="saga", random_state=RANDOM_SEED)
    global_model.fit(np.zeros((2, N_FEATURES)), [0, 1])
    global_params = (global_model.coef_.copy(), global_model.intercept_.copy())

    for fl_round in range(1, NUM_ROUNDS + 1):
        client_params_list: list[tuple[np.ndarray, np.ndarray]] = []
        client_sizes: list[int] = []
        for client_id in range(3):
            X_train, X_test, y_train, y_test = splits[client_id]
            client_model = _make_client_model(global_params[0], global_params[1])
            client_model.fit(X_train, y_train)
            acc = accuracy_score(y_test, client_model.predict(X_test))
            FL_PROGRESS["clients"][client_id]["history"].append({"round": fl_round, "accuracy": round(float(acc), 4)})
            client_params_list.append((client_model.coef_, client_model.intercept_))
            client_sizes.append(len(X_train))

        avg_coef, avg_intercept = _fedavg_params(client_params_list, client_sizes)
        global_params = (avg_coef, avg_intercept)
        global_model.coef_ = avg_coef
        global_model.intercept_ = avg_intercept

        all_X_test = np.vstack([s[1] for s in splits])
        all_y_test = np.hstack([s[3] for s in splits])
        global_acc = accuracy_score(all_y_test, global_model.predict(all_X_test))
        FL_PROGRESS["global_history"].append({"round": fl_round, "accuracy": round(float(global_acc), 4)})
        FL_PROGRESS["round"] = fl_round
        joblib.dump(global_model, str(FL_MODEL_PATH))
        print(f"[fl] round {fl_round}/{NUM_ROUNDS} global={global_acc:.4f}")

    FL_PROGRESS["running"] = False
    print("[fl] simulation complete")
```

Create backend/ml/fl_inference.py with the following implementation.

This module maintains a module-level _FL_MODEL reference that is atomically reassigned by reload_model(). The atomic Python reference assignment (GIL-safe in CPython) means no locking is needed. The 60s reload task calls reload_model() after each sleep cycle.

```python
"""
AegisFlow FL inference module.

Module-level _FL_MODEL is reloaded every 60 seconds by the _fl_reload_loop
background task in ws.py. Reference assignment is GIL-atomic in CPython.
predict_with_fl() supplements (does not replace) the XGBoost anomaly score.
"""
import joblib
import numpy as np
from pathlib import Path
from sklearn.linear_model import LogisticRegression

_FL_MODEL: LogisticRegression | None = None
_FL_MODEL_PATH = Path(__file__).parent / "fl_model.pkl"


def reload_model(path: str | None = None) -> None:
    """
    Load or reload FL global model from disk.
    GIL-safe: _FL_MODEL reference assignment is atomic in CPython.
    Called by _fl_reload_loop every 60 seconds.
    """
    global _FL_MODEL
    target = Path(path) if path else _FL_MODEL_PATH
    if target.exists():
        _FL_MODEL = joblib.load(str(target))


def predict_with_fl(features: np.ndarray) -> float | None:
    """
    Run FL model inference on a 9-feature vector.

    Args:
        features: 1D float32 ndarray of shape (9,) matching XGBoost feature schema

    Returns:
        Anomaly probability [0.0, 1.0] or None if model not yet loaded.
        None is returned until the first 60s reload completes after round 1.
    """
    if _FL_MODEL is None:
        return None
    prob = _FL_MODEL.predict_proba(features.reshape(1, -1))[0][1]
    return float(prob)
```
