```python
async def _fl_loop():
    """
    Run FL simulation ONCE at backend startup, offloaded to ThreadPoolExecutor.
    ThreadPoolExecutor prevents blocking the asyncio event loop during CPU-bound training.
    run_fl_simulation() is synchronous — must NOT be awaited directly.
    """
    from backend.ml.fl_train import run_fl_simulation
    loop = asyncio.get_event_loop()
    try:
        await loop.run_in_executor(_fl_executor, run_fl_simulation)
    except Exception as e:
        print(f"[fl] simulation error: {e}")
    # Task completes — FL runs once per backend startup


async def _fl_reload_loop():
    """
    Reload FL global model from disk every 60 seconds (FL-03).
    Picks up fl_model.pkl written by run_fl_simulation() after each round.
    Model reference reassignment is GIL-atomic in CPython — no lock needed.
    """
    from backend.ml import fl_inference
    model_path = __import__('pathlib').Path("backend/ml/fl_model.pkl")
    while True:
        await asyncio.sleep(60)
        if model_path.exists():
            try:
                fl_inference.reload_model(str(model_path))
                print("[fl] reloaded global FL model")
            except Exception as e:
                print(f"[fl] reload error: {e}")


@router.get("/api/fl/status")
async def get_fl_status():
    """Return current FL simulation progress and accuracy history for frontend polling."""
    from backend.ml.fl_train import FL_PROGRESS
    return JSONResponse(FL_PROGRESS)
```
