NodeStatus = Literal["normal", "warning", "critical"]

**backend/routers/ws.py** — WebSocket broadcast + storm endpoint:
```python
import asyncio
import random
from fastapi import APIRouter, WebSocket, WebSocketDisconnect
from fastapi.responses import JSONResponse
from backend.state import NODE_STATES, simulate_sensor_tick

router = APIRouter()

# Connected WebSocket clients
_clients: set[WebSocket] = set()

@router.websocket("/ws/nodes")
async def ws_nodes(websocket: WebSocket):
    await websocket.accept()
    _clients.add(websocket)
    try:
        while True:
            # Keep connection alive — broadcast loop drives messages
            await asyncio.sleep(30)
    except WebSocketDisconnect:
        _clients.discard(websocket)

@router.on_event("startup")
async def start_broadcast_loop():
    asyncio.create_task(_broadcast_loop())

async def _broadcast_loop():
    while True:
        simulate_sensor_tick()
        payload = [n.model_dump() for n in NODE_STATES.values()]
        dead = set()
        for ws in _clients.copy():
            try:
                await ws.send_json(payload)
            except Exception:
                dead.add(ws)
        _clients -= dead
        await asyncio.sleep(1)

@router.post("/api/storm")
async def inject_storm():
    """Inject a fault: pick 1 generator as epicenter, set 4 nearby nodes to warning/critical."""
    import random
    node_ids = list(NODE_STATES.keys())
    epicenter_id = random.choice([nid for nid in node_ids if NODE_STATES[nid].type == "generator"])
    affected = random.sample([nid for nid in node_ids if nid != epicenter_id], k=4)
    NODE_STATES[epicenter_id].status = "critical"
    NODE_STATES[epicenter_id].load = min(99.0, NODE_STATES[epicenter_id].load + 20)
    NODE_STATES[epicenter_id].frequency = 59.1
    for nid in affected[:2]:
        NODE_STATES[nid].status = "warning"
        NODE_STATES[nid].load = min(92.0, NODE_STATES[nid].load + 15)
    for nid in affected[2:]:
        NODE_STATES[nid].status = "critical"
        NODE_STATES[nid].load = min(97.0, NODE_STATES[nid].load + 18)
    return JSONResponse({"injected": True, "epicenter": epicenter_id, "affected": affected})
```

Note: `@router.on_event("startup")` is deprecated in FastAPI 0.115 — use `lifespan` instead. Attach the broadcast loop via a `lifespan` context manager on the FastAPI app in `main.py`. Update main.py to use:
```python
from contextlib import asynccontextmanager

@asynccontextmanager
async def lifespan(app: FastAPI):
    asyncio.create_task(_broadcast_loop())
    yield

app = FastAPI(title="AegisFlow API", lifespan=lifespan)
```

Import `_broadcast_loop` from `backend.routers.ws` in `main.py` after the router is defined. Remove the `@router.on_event("startup")` from ws.py — the broadcast task is started by the lifespan in main.py.
